SELECT  J.JOB_ID, J.MIN_SALARY, J.MAX_SALARY, H.DEPARTMENT_ID
FROM JOBS J, JOB_HISTORY H
WHERE J.JOB_ID = H.JOB_ID AND J.JOB_ID = 'IT_PROG';
