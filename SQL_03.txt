--3일차

--LOWER(), UPPER(), INITCAP()
SELECT FIRST_NAME, LOWER(FIRST_NAME), UPPER(FIRST_NAME), INITCAP(FIRST_NAME) 
FROM EMPLOYEES;


--SUBSTR() : 문자열에서 지정된 길이 숫자만큼 추출
--1번째 위치에서 3글자 추출:
SELECT FIRST_NAME, SUBSTR(FIRST_NAME, 1, 3)
FROM EMPLOYEES;


--3번째 위치에서 2글자 추출:
SELECT FIRST_NAME, SUBSTR(FIRST_NAME, 3, 2)
FROM EMPLOYEES;


--REPLACE() : 문자열을 찾아서 바꾸는 함수
SELECT JOB_ID, REPLACE(JOB_ID, 'VP', 'MVP') AS 체인지
FROM EMPLOYEES;


--CONCAT() : 두 문자열을 하나로 합치는 함수
SELECT FIRST_NAME, LAST_NAME, CONCAT(FIRST_NAME, LAST_NAME) AS FULL_NAME
FROM EMPLOYEES;


--두 단어 사이 빈칸 생성:
SELECT FIRST_NAME, LAST_NAME, CONCAT(FIRST_NAME, CONCAT(' ', LAST_NAME)) AS FULL_NAME
FROM EMPLOYEES;


--LENGTH() : 길이, INSTR() : 문자열 위치
SELECT FIRST_NAME, LENGTH(FIRST_NAME) AS 길이, INSTR(FIRST_NAME, 'a') AS a의위치
FROM EMPLOYEES;


--LPAD(), RPAD() : 특정 문자를 왼쪽, 오른쪽에서 채워주는 함수
SELECT LPAD(FIRST_NAME, 10, '*'), RPAD(FIRST_NAME, 10, '#')
FROM EMPLOYEES;


--LTRIM(), RTRIM(), TRIM()
SELECT JOB_ID, LTRIM(JOB_ID, 'A'), RTRIM(JOB_ID, 'T')
FROM EMPLOYEES;


--TEST1: employees 테이블에서 first_name 첫1문자와 last_name을 중간에 공백을 두고 하나로 결합하여 조회
SELECT FIRST_NAME, LAST_NAME, CONCAT(SUBSTR(FIRST_NAME, 1, 1), CONCAT(' ', LAST_NAME))
FROM EMPLOYEES;


--TEST2: employees 테이블에서 first_name과 last_name의 길이를 합쳐서 조회
SELECT FIRST_NAME, LAST_NAME, LENGTH(CONCAT(FIRST_NAME, LAST_NAME))
FROM EMPLOYEES;


--TEST3: employees 테이블에서 이름과 월급을 출력하는데 숫자0을 *로 출력
SELECT FIRST_NAME, LAST_NAME, REPLACE(SALARY, '0', '*')
FROM EMPLOYEES;


--TEST4: employees 테이블에서 이름과 월급을 출력하는데 숫자1번부터 3번까지는 *로 출력(정규표현식: REGEXP_REPLACE 사용!)
SELECT FIRST_NAME, LAST_NAME, REGEXP_REPLACE(SALARY, '[1-3]', '*')
FROM EMPLOYEES;


--날짜 출력:
SELECT SYSDATE AS 오늘, SYSDATE+1 AS 내일, SYSDATE-1 AS 어제
FROM DUAL;


--ADD_MONTHS()
SELECT HIRE_DATE, ADD_MONTHS(HIRE_DATE, 2) --2달 더하기
FROM EMPLOYEES;


--NVL() : NULL값을 특정값으로 치환
SELECT DEPARTMENT_NAME, MANAGER_ID, NVL(MANAGER_ID, 100)
FROM DEPARTMENTS;


--NVL2(): NULL값이 아닌값을 첫번째값으로, NULL값을 두번째값으로 치환
SELECT DEPARTMENT_NAME, MANAGER_ID, NVL2(MANAGER_ID, '관리자있음', '관리자없음')
FROM DEPARTMENTS;
